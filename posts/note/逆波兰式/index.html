class Solution {
public:
    int evalRPN(vector<string>& tokens) {
        if (tokens.empty()) return 0;
        stack<long long int> st;
        for (auto token : tokens) {
            if (token == "+" || token == "-" || token == "*" || token == "/") {
                int b = st.top();
                st.pop();
                int a = st.top();
                st.pop(); 
                if (token == "+") {
                    st.push(a+b);
                } else if (token == "-") {
                    st.push(a-b);
                } else if (token == "*") {
                    st.push(a*b);
                } else if (token == "/") {
                    st.push(a/b);
                }
            } else {
                st.push(stoi(token));
            }
        }
        return (int)st.top();
    }
};
