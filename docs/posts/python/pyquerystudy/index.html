伪类选择器


doc = pq(html)

li = doc("li:first-child")

print(li.text())

li = doc("li:last-child")

print(li.text())

li = doc("li:nth-child(2)")

print(li.text())

li = doc("li:nth-child(3n)")

print(li.text())

li = doc("li:gt(3)")#取第5个,即第4个以后,第四个下标为3

print(li.text())

li = doc("li:contains(second)")#搜索文本
