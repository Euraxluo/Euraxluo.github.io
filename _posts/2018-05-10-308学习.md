+ 代码：

```java

import java.io.File;

import java.io.IOException;



public class homework508 {

	public static void main(String[] args) {

	find();

	}

	public static void create() throws IOException {



		//在当前目录创建文件

		File file = new File("test.txt");

		boolean flag = file.createNewFile();

		if(flag)

		{

			System.out.println("创建成功");

		}else {

			System.out.println("文件已存在");

		}



		//创建一个目录

		File newdir = new File("test");

		flag = newdir.mkdir();

		if(flag)

		{

			System.out.println("创建成功");

		}else {

			System.out.println("创建失败");

		}



		//创建多级目录

		File dirs = new File("F:"+ File.separator+"eclipse-workspace"+File.separator+"xc-java"+File.separator+"src"+File.separator +"homework"+File.separator +"test.txt");



		flag = dirs.mkdirs();

		if(flag)

		{

			System.out.println("创建成功");

		}else {

			System.out.println("创建失败");

		}

	}

	public static void delete()

	{

		//同目录删除

		File file = new File("test.txt");

		boolean flag = file.delete();

		System.out.println(flag);

		if(flag)

		{

			System.out.println("删除成功");

		}else {

			System.out.println("删除失败");

		}



	}

	public static void find()

	{

		File file = new File("."+File.separator+"homework428.java");

		System.out.println("文件名："+file.getName());

		System.out.println("文件大小："+file.length()+"字节");

	}

}

```

import java.util.Scanner;

import javax.script.ScriptEngine;

import javax.script.ScriptEngineManager;

public class jisuqnqi {

    public static void main(String[] args) {

        Scanner input = new Scanner(System.in);

        System.out.println("请输入一个表达式：");

        String str = input.nextLine();



        //new一个脚本引擎管理对象

        ScriptEngineManager manager = new ScriptEngineManager();

        //获取JS引擎

        ScriptEngine engine = manager.getEngineByName("js");

        //计算表达式 输出

        try {

            System.out.println("答案：" + engine.eval(str));

        } catch (Exception e) {

            System.out.println(e);

        }

    }



}
